      #include <stdio.h>
      #include <stdlib.h>
      
      #define MAX_SIZE 100
      
      int main() {
        int arr[MAX_SIZE], freq[MAX_SIZE];
        int n, sum = 0, mean, median, mode;
        int i, j, max_freq = 0, freq_count = 0;
      
        printf("Enter the number of elements in the array: ");
        scanf("%d", &n);
      
        printf("Enter %d elements in the array:\n", n);
        for (i = 0; i < n; i++) {
          scanf("%d", &arr[i]);
          sum += arr[i];
        }
      
        // Mean calculation
        mean = sum / n;
      
        // Median calculation
        for (i = 0; i < n; i++) {
          for (j = i + 1; j < n; j++) {
            if (arr[i] > arr[j]) {
              int temp = arr[i];
              arr[i] = arr[j];
              arr[j] = temp;
            }
          }
        }
        if (n % 2 == 0) {
          median = (arr[n/2] + arr[n/2 - 1]) / 2;
        } else {
          median = arr[n/2];
        }
      
        
        for (i = 0; i < n; i++) {
          freq[i] = 0;
        }
        for (i = 0; i < n; i++) {
          for (j = 0; j < n; j++) {
            if (arr[i] == arr[j]) {
              freq[i]++;
            }
          }
        }
        for (i = 0; i < n; i++) {
          if (max_freq < freq[i]) {
            max_freq = freq[i];
            mode = arr[i];
            freq_count = 1;
          } else if (max_freq == freq[i]) {
            freq_count++;
            if (freq_count == 2) {
              mode = 0;
            }
          }
        }
      
        // Print results
        printf("Mean: %d\n", mean);
        printf("Median: %d\n", median);
        if (mode != 0) {
          printf("Mode: %d\n", mode);
        } else {
          printf("No mode\n");
        }
      
        return 0;
      }
